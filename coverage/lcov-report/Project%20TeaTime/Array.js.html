class Array{ 
  constructor(){ 
    this.data =[];
    this.length = 0;
  } 
 
  push(element) { 
    this.data[this.length] = element; 
    this.length++; 
    return this.length; 
  } 
 
 
  
  pop(element){  
    this.data[this.length-1] = element; 
    delete this.data[this.length-1]; 
    this.length--; 
    return this.data;
  }

      this.data[i]=this.data[i-1];

    this.data[index]=item;

    return this.data;
